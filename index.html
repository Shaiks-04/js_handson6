    4.     var Perfect_Check = (N) => 
    {
      if(N===1){
        return 'YES';
      }     
      let sum=0;
      for(let i=0;i<=(N/2);i++){
        if(N%i===0){
          sum=sum+i;
        }
      }
      if(sum!==0 && N===sum){
        return 'YES';
      }
      else{
        return 'NO';
      }
    };
